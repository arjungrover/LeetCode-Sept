/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {
    public int sumRootToLeaf(TreeNode root) {
        TreeNode ans = new TreeNode(0);
        dfs(root, root.val , ans);
        return ans.val;
    }
    
    public void dfs(TreeNode node, int val, TreeNode ans){
        //leaf property
        if(node.left==null && node.right==null)
        {
            ans.val+=val;
            return;
        }
        if(node.left!=null)
            dfs(node.left, (val*2)+node.left.val, ans);
        if(node.right!=null)
            dfs(node.right, (val*2)+node.right.val, ans);
        
    }
}
